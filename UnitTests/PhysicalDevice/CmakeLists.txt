cmake_minimum_required(VERSION 3.19)
cmake_policy(SET CMP0074 NEW)

Fetch_GoogleTest()

set(OCRA_TEST_PHYSICAL_DEVICE_SOURCE_FILES
  ${CMAKE_CURRENT_SOURCE_DIR}/src/PhysicalDevice.cpp)
add_executable(OCRA-UnitTests-PhysicalDevice
  ${OCRA_TEST_PHYSICAL_DEVICE_SOURCE_FILES})
target_include_directories(OCRA-UnitTests-PhysicalDevice PRIVATE
  "${CMAKE_CURRENT_SOURCE_DIR}/include")

if(MSVC_IDE)
  set_property(TARGET OCRA-UnitTests-PhysicalDevice PROPERTY
    VS_DEBUGGER_ENVIRONMENT "${OCRA_DEBUGGER_ENV}")
endif()

target_link_libraries(
  OCRA-UnitTests-PhysicalDevice PRIVATE
    GTest::GTest
    OCRA::API::Interface
    OCRA::API::Implementation)

add_test(PhysicalDeviceTests OCRA-UnitTests-PhysicalDevice)

if(MSVC_IDE)
  # Run macro
  GroupSources(src)
  GroupSources(include)
endif()
