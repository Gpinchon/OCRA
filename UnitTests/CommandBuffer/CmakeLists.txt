cmake_minimum_required(VERSION 3.19)
cmake_policy(SET CMP0074 NEW)

set(OCRA_TEST_COMMAND_BUFFER_SOURCE_FILES
  ${CMAKE_CURRENT_SOURCE_DIR}/src/CommandBuffer.cpp)
add_executable(OCRA-UnitTests-CommandBuffer
  ${OCRA_TEST_COMMAND_BUFFER_SOURCE_FILES})
target_include_directories(OCRA-UnitTests-CommandBuffer PRIVATE
  "${CMAKE_CURRENT_SOURCE_DIR}/include")

if(MSVC_IDE)
  set_property(TARGET OCRA-UnitTests-CommandBuffer PROPERTY
    VS_DEBUGGER_ENVIRONMENT "${OCRA_DEBUGGER_ENV}")
endif()

target_link_libraries(OCRA-UnitTests-CommandBuffer OCRA-UnitTests-Lib-Common OCRA::API::Interface OCRA::API::Implementation)

if(MSVC_IDE)
  # Run macro
  GroupSources(src)
  GroupSources(include)
endif()
