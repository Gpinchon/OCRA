cmake_minimum_required(VERSION 3.19)
cmake_policy(SET CMP0074 NEW)

Fetch_GoogleTest()

set(OCRA_TEST_QUEUE_SOURCE_FILES
  ${CMAKE_CURRENT_SOURCE_DIR}/src/Queue.cpp)
add_executable(OCRA-UnitTests-Queue
  ${OCRA_TEST_QUEUE_SOURCE_FILES})
target_include_directories(OCRA-UnitTests-Queue PRIVATE
  "${CMAKE_CURRENT_SOURCE_DIR}/include")

if(MSVC_IDE)
  set_property(TARGET OCRA-UnitTests-Queue PROPERTY
    VS_DEBUGGER_ENVIRONMENT "${OCRA_DEBUGGER_ENV}")
endif()

target_link_libraries(
  OCRA-UnitTests-Queue PRIVATE
    GTest::GTest
    OCRA-UnitTests-Lib-Common
    OCRA::API::Interface
    OCRA::API::Implementation)

if(MSVC_IDE)
  # Run macro
  GroupSources(src)
  GroupSources(include)
endif()
